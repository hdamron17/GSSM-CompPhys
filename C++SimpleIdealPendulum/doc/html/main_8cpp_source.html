70 vector<tuple<double, double>> exact_damped_driven(double theta0,
71  double dt, double end_t, double nat_freq,
72  double friction_coef, double driving_freq, double driving_torque,
73  bool plot_x_vs_time) {
74  vector<tuple<double,double>> data; //(t, theta)
75  data.push_back(make_tuple(0, theta0));
76  while(get<0>(data.back()) < end_t) {
77  tuple<double,double> last = data.back();
78  double t_new = get<0>(last) + dt;
79  if(is_close(friction_coef, 2 * nat_freq)) {
80  //Critically damped
81  double temp = friction_coef * t_new / 2;
82  double theta_new = theta0 * (1 + temp) * exp(-temp);
83  data.push_back(make_tuple(t_new, theta_new));
84  } else if(friction_coef < 2 * nat_freq) {
85  //Underdamped
86  double temp = sqrt(pow(nat_freq,2) - pow(friction_coef,2) / 4);
87  double theta_new = theta0 * exp(-friction_coef * t_new / 2) * \
88  (cos(temp * t_new) + friction_coef / 2 / temp \
89  * sin(temp * t_new));
90  data.push_back(make_tuple(t_new, theta_new));
91  } else {
92  //Overdamped
93  double temp = sqrt(pow(friction_coef,2) / 4 - pow(nat_freq,2));
94  double theta_new = theta0 * exp(-friction_coef * t_new / 2) * \
95  (cosh(temp * t_new) + friction_coef / 2 / temp \
96  * sinh(temp * t_new));
97  data.push_back(make_tuple(t_new, theta_new));
98  }
99  }

102  Gnuplot gp;

124 tuple<double,double,double> one_step(tuple<double,double,double> prev,double dt,
125  double nat_freq, double friction_coef, double driving_freq,
126  double driving_torque, bool linear=true) {
127  double theta_new, ang_v_new;
128  double time_new = get<0>(prev) + dt;
129  if(linear) {
130  double ai = dw_dt(get<1>(prev), get<2>(prev), get<0>(prev), nat_freq,
131  friction_coef, driving_freq, driving_torque);
132  theta_new = get<1>(prev) + get<2>(prev) * dt + 0.5 * ai \
133  * pow(dt, 2);
134  ang_v_new = (get<2>(prev) + 0.5 * (ai - pow(nat_freq, 2) * theta_new \
135  + driving_torque * sin(driving_freq * time_new) ) * dt) \
136  / (1 + friction_coef / 2 * dt);
137  } else {
138  if(friction_coef != 0 || driving_freq != 0 || driving_torque != 0)
139  cerr << "Nonlinear pendulum cannot have damping or driving\n";
140  double ai = dw_dt_nonlinear(get<1>(prev), nat_freq);
141  theta_new = get<1>(prev) + get<2>(prev) * dt + 0.5 * ai \
142  * pow(dt, 2);
143  double af = dw_dt_nonlinear(theta_new, nat_freq);
144  ang_v_new = get<2>(prev) + 0.5 * (ai + af) * dt;
145  }
146  while(theta_new < -M_PI) {
147  theta_new += 2 * M_PI; //add rotations until positive
148  }
149  while(theta_new > M_PI) {
150  theta_new -= 2 * M_PI; //subtract rotations until negative
151  }
152  return make_tuple(time_new, theta_new, ang_v_new);
153 }

225 bool _stable(deque<tuple<double,double>> turns, double epsilon) {

228  }

234  || (i % 2 == 1 && !is_close(get<0>(turn), peak2, epsilon)) ) { //case: i is odd
235  return false;

237  i++;
238  }

247 tuple<double,double> _calc_ampl_freq(deque<tuple<double,double>> turns) {
248  //Note: deque should be even size but is not required
249  int i = 0;
250  double prev_time;
251  double peak_total1 = 0; //sums either mins or maxes
252  double peak_total2 = 0; //sums the opposite of peak_total1
253  double time_total = 0; //sums periods
254  for(tuple<double,double> turn : turns) {
255  if(i % 2 == 0) {
256  peak_total1 += get<0>(turn);
257  prev_time = get<1>(turn);
258  } else {
259  peak_total2 += get<0>(turn);
260  time_total += get<1>(turn) - prev_time;
261  }
262  i++;
263  }
264  int size = turns.size();
265  double avg_peak1 = peak_total1 / (size / 2 + size % 2); //avg for first set
266  double avg_peak2 = peak_total2 / (size / 2); //avg for second set
267  double avg_time = time_total / (size / 2); //avg period
268  return make_tuple( abs(avg_peak1 - avg_peak2) / 2, 1 / avg_time );
269 }

285 tuple<double,double> _ampl_freq(double theta0,
286  double ang_v0, double dt, double nat_freq,
287  double friction_coef, double driving_freq, double driving_torque,
288  bool linear=true, double linear_time=20, int loops_precision=10) {
289  int lin_max_steps = int(linear_time / dt) + 1;
290  deque<tuple<double,double>> turns; //positions and times of mins and maxes
291  tuple<double,double> initializer = make_tuple(theta0 + 1, 0); //value to fill deque before looping
292  for(int i = 0; i < 2*loops_precision; i++) {
293  turns.push_back(initializer);
294  }
295  int same_count = 0; //counts the number of points which are the same in a row
296  bool direction = true;
297  tuple<double,double,double> step = make_tuple(0, theta0, ang_v0);
298 
299  while(true) {
300  step = one_step(step, dt, nat_freq, friction_coef, driving_freq, \
301  driving_torque, linear);
302  double new_ang_v = get<2>(step);
303  if(is_close(new_ang_v, 0)) {
304  //case: next_step has same position as previousd
305  same_count++;
306  } else if(
307  //case: next step is positive and direction is negative
308  (new_ang_v > 0 && !direction)
309  //case: ang_v is negative and direction is positive
310  || (new_ang_v < 0 && direction) ) {
311  direction = !direction; //reverse direction
312  turns.pop_front(); //remove oldest min/max
313  turns.push_back( make_tuple(get<1>(step), get<0>(step)) ); //new pos/time
314  if(_stable(turns, dt / 2)) { //precision capabilities based on dt
315  return _calc_ampl_freq(turns); //calculates amplitude and frequency
316  }
317  }
318  if(same_count > lin_max_steps) {
319  return make_tuple(0, 0); //linear has no amplitude or position
320  }
321  }
