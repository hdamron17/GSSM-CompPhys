104 tuple<double,double,double> one_step(tuple<double,double,double> prev,double dt,
105  double nat_freq, double friction_coef, double driving_freq,
106  double driving_torque, bool linear=true, bool bounded=true) {
107  double theta_new, ang_v_new;
108 
109  double time_new = get<0>(prev) + dt;
110  double ai = dw_dt(get<1>(prev), get<2>(prev), get<0>(prev), nat_freq,
111  friction_coef, driving_freq, driving_torque, linear);
112  theta_new = get<1>(prev) + get<2>(prev) * dt + 0.5 * ai \
113  * pow(dt, 2);
114  ang_v_new = (get<2>(prev) + 0.5 * (ai - pow(nat_freq, 2) * \
115  (linear ? theta_new : sin(theta_new)) + driving_torque * \
116  sin(driving_freq * time_new) ) * dt) / \
117  (1 + friction_coef / 2 * dt);
118  if(bounded) {
119  while(theta_new < -M_PI) {
120  theta_new += 2 * M_PI; //add rotations until positive
121  }
122  while(theta_new > M_PI) {
123  theta_new -= 2 * M_PI; //subtract rotations until negative
124  }
125  }
126  return make_tuple(time_new, theta_new, ang_v_new);
127 }
